# 使用C++
``` C++
#include<iostream>
using namespace std;
int main()
{
    int num;
    cin>>num;
    if(num==0){
        cout<<'a'<<endl;
        return 0;
    }
    int a[9];//用于倒序存放num
    char b[9];//用于输出
    for(int i=0;i<9;i++)
        a[i]=-1;
    int n=num;
    for(int i=0;i<9 && n>0;i++){
        a[i]=n%10;
        n/=10;
    }
    for(int i=8;i>=0;i--){
        if(a[i]>0){//不是-1和0
            printf("%c",'a'+a[i]);
            switch(i){
                case 8:cout<<"Y";break;
                case 7:cout<<"Q";break;
                case 6:cout<<"B";break;
                case 5:cout<<"S";break;
                case 4:cout<<"W";break;
                case 3:cout<<"Q";break;
                case 2:cout<<"B";break;
                case 1:cout<<"S";break;
            }
        }
        else if(a[i]==0 && i-1<0)
            continue;
        else if(a[i]==0 && a[i-1]!=0)
            printf("%c",'a'+a[i]);
        if(i==4 && a[i]==0)
            cout<<"W";
    }
    
    return 0;
}
```
